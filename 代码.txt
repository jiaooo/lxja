#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int* a = new int[n];
 
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];//输入数据
	}
	sort(a, a + n);//将数据由低到高排序,a为首地址，a+n为尾地址
 
	for (int i = 0; i < n; i++)
	{
		cout << a[i] << " ";//先输出a[i]的值再输出“  ”
	}
	int i = 0;
	int MaxCount = 1;//将最大值赋为1
	int index = 0;
	while (i < n - 1)//遍历求出众数
	{
		int count = 1;//出现的次数
		int j ;
		for (j = i; j < n - 1; j++)
		{
			if (a[j] == a[j + 1])//存在连续两个数相等，则众数+1
			{
				count++;
			}
			else
			{
				break;
			}
		}
		if (MaxCount < count)
		{
			MaxCount = count;//当前最大众数
			index = j;//当前众数标记位置
		}
		++j;
		i = j;//位置后移到下一个未出现的数字
	}
	cout << a[index] << " " << MaxCount;
	return 0;
}
